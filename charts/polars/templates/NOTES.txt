1. Get the Polars Scheduler URL:
{{- if contains "NodePort" .Values.scheduler.services.scheduler.type }}
  ```bash
  export SCHEDULER_NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "polars.scheduler.fullname" . }})
  export SCHEDULER_NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$SCHEDULER_NODE_IP:$SCHEDULER_NODE_PORT
  ```
{{- else if contains "LoadBalancer" .Values.scheduler.services.scheduler.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch its status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "polars.scheduler.fullname" . }}'
  ```bash
  export SCHEDULER_SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "polars.scheduler.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SCHEDULER_SERVICE_IP:5051
  ```
{{- else if contains "ClusterIP" .Values.scheduler.services.scheduler.type }}
  kubectl --namespace {{ .Release.Namespace }} port-forward service/{{ include "polars.scheduler.fullname" . }} 5051:5051
{{- end }}

Use Polars on-premises with Python:
  ```python
  import os
  import polars as pl
  import polars_cloud as pc

  # Directly connect to the cluster
  {{ include "polars.notes.clusterContext" . }}
  result = (
    pl.LazyFrame(
      {
        "a": [1, 2, 3],
        "b": [4, 4, 5],
      }
    )
    .with_columns(
      pl.col("a").max().over("b").alias("c"),
    )
    .remote(ctx)
    .distributed()
    {{- if not (include "polars.isAnonymousResultsEnabled" .) }}
    .sink_parquet("s3://your-bucket/path/to/output")
    {{- else }}
    .execute()
    {{- end }}
  )

  print(result.head)
  ```
