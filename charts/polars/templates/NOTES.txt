1. Get the Polars Scheduler URL:
{{- if contains "NodePort" .Values.scheduler.services.scheduler.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "polars.scheduler.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.scheduler.services.scheduler.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch its status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "polars.scheduler.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "polars.scheduler.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:5051
{{- else if contains "ClusterIP" .Values.scheduler.services.scheduler.type }}
  kubectl --namespace {{ .Release.Namespace }} port-forward service/{{ include "polars.scheduler.fullname" . }} 5051:5051
{{- end }}

Use Polars on-premises with Python:
  ```python
  import os
  import polars as pl
  import polars_cloud as pc

  # Directly connect to the cluster
  ctx = pc.ClusterContext(compute_address={{- if contains "NodePort" .Values.scheduler.services.scheduler.type }}os.environ["NODE_IP"]{{- else if contains "LoadBalancer" .Values.scheduler.services.scheduler.type }}os.environ["SERVICE_IP"]{{- else if contains "ClusterIP" .Values.scheduler.services.scheduler.type }}"127.0.0.1"{{- end }}{{- if contains "NodePort" .Values.scheduler.services.scheduler.type }}, scheduler_port=int(os.environ["NODE_PORT"]){{- end }}, insecure=True)
  result = (
    pl.LazyFrame(
      {
        "a": [1, 2, 3],
        "b": [4, 4, 5],
      }
    )
    .with_columns(
      pl.col("a").max().over("b").alias("c"),
    )
    {{- if not (include "polars.isAnonymousResultsEnabled" .) }}
    .sink_parquet("s3://your-bucket/path/to/output.parquet", sink_to_single_file=True)
    {{- end }}
    .remote(ctx)
    .execute()
  )

  print(result.head)
  ```
